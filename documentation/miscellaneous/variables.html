src/app/Core/Guards/auth.guard.ts

authGuard
Type : CanActivateFn
Default value : (route, state) => { const router = inject(Router); const _auth = inject(AuthServiceService); const token = localStorage.getItem('token'); const expiry = localStorage.getItem('tokenExpiry'); if (token && expiry) { const now = new Date(); const expiryDate = new Date(expiry); if (now > expiryDate) { localStorage.clear(); return router.parseUrl('/login'); } } if (_auth.isAuthenticated() || token) { return true; } return router.parseUrl('/login'); }

src/app/Core/Interceptors/auth.interceptor.ts

authInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { const token = localStorage.getItem('token'); if (token) { const clonedReq = req.clone({ setHeaders: { Authorization: `Bearer ${token}` }, }); return next(clonedReq); } return next(req); }

src/app/Core/Interceptors/loading.interceptor.ts

loadingInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { const spinner = inject(NgxSpinnerService); spinner.show(); return next(req).pipe( finalize(() => { spinner.hide(); }) ); }
